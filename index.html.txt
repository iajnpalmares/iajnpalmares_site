<script>
  // Gerenciar troca de páginas internas
  const buttons = document.querySelectorAll('nav button');
  const sections = document.querySelectorAll('main section');

  buttons.forEach(button => {
    button.addEventListener('click', () => {
      const page = button.getAttribute('data-page');

      // Atualiza o botão ativo
      buttons.forEach(btn => {
        btn.classList.toggle('active', btn === button);
        btn.setAttribute('aria-current', btn === button ? 'page' : 'false');
      });

      // Mostra a seção certa
      sections.forEach(sec => {
        sec.classList.toggle('active', sec.id === page);
      });

      // Scrolla para topo do main
      window.scrollTo({top: 0, behavior: 'smooth'});
    });
  });

  // Scroll suave para topo
  function scrollToTop() {
    window.scrollTo({top: 0, behavior: 'smooth'});
  }

  // Mostrar/esconder botão topo
  window.addEventListener('scroll', () => {
    const btnTop = document.getElementById('btn-top');
    if(window.scrollY > 300){
      btnTop.style.display = 'block';
    } else {
      btnTop.style.display = 'none';
    }
  });

  // Botão devocional simulado (você pode integrar Google Sheets aqui)
  function carregarDevocional(){
    const container = document.getElementById('devocional-container');
    container.textContent = '“Entrega o teu caminho ao Senhor; confia nele, e ele o fará.” — Salmo 37:5';
  }

  // Envio simulado do formulário oração
  function enviarOracao(event){
    event.preventDefault();
    const form = event.target;
    const status = document.getElementById('status-oracao');
    const pedido = form.pedido.value.trim();
    if(pedido.length === 0){
      status.textContent = 'Por favor, escreva seu pedido de oração.';
      return false;
    }
    status.textContent = 'Pedido de oração enviado com sucesso! Obrigado.';
    form.reset();
    return false;
  }

  // Envio simulado do formulário visita
  function solicitarVisita(event){
    event.preventDefault();
    const form = event.target;
    const status = document.getElementById('status-visita');
    const nome = form.nome.value.trim();
    const endereco = form.endereco.value.trim();
    const telefone = form.telefone.value.trim();
    if(!nome || !endereco || !telefone){
      status.textContent = 'Por favor, preencha todos os campos.';
      return false;
    }
    status.textContent = 'Solicitação de visita enviada com sucesso! Entraremos em contato.';
    form.reset();
    return false;
  }
</script>
